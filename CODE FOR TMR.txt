CODE FOR TMR

CREATE TABLE `airline_management`.`passenger` (`id` INT NOT NULL AUTO_INCREMENT , `user_id` INT NOT NULL , `gender` ENUM('Male', 'Female', 'Other') NOT NULL , `d_o_b` DATE NOT NULL , `passport_no` VARCHAR(20) NOT NULL , `NID` VARCHAR(20) NOT NULL , `status` ENUM('Active', 'Inactive', 'Banned') NOT NULL , `phone` VARCHAR(20) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

ALTER TABLE `passenger` ADD FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE ON UPDATE CASCADE;


EMPLOYEE


CREATE TABLE `airline_management`.`employee` (`user_id` INT NOT NULL AUTO_INCREMENT , `hire_date` DATE NOT NULL , `base_salary` DECIMAL(15,2) NOT NULL , `phone` VARCHAR(20) NOT NULL , PRIMARY KEY (`user_id`)) ENGINE = InnoDB;
ALTER TABLE `employee` ADD FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE ON UPDATE CASCADE;

ADMIN

CREATE TABLE `airline_management`.`admin` (`id` INT NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
ALTER TABLE `admin` ADD FOREIGN KEY (`id`) REFERENCES `employee`(`user_id`) ON DELETE CASCADE ON UPDATE CASCADE;


CREW
CREATE TABLE `airline_management`.`crew` (`id` INT NOT NULL , `duty_role` VARCHAR(50) NOT NULL , `start_time` DATETIME NOT NULL , `end_time` DATETIME NOT NULL , `status` ENUM('On Duty', 'Off Duty', 'On Leave', 'Standby') NOT NULL , `type` VARCHAR(30) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;


ALTER TABLE `crew` ADD FOREIGN KEY (`id`) REFERENCES `employee`(`user_id`) ON DELETE CASCADE ON UPDATE CASCADE;
